Get-ChildItem -Path .\routes -Recurse -Filter *.js |
  ForEach-Object {
    "`n=== FILE: $($_.FullName) ===`n"
    Get-Content -Path $_.FullName -Raw
  } |
  Out-File routes-content.txt -Encoding UTF8
